```python
# src/pxh/race.py
"""Autonomous track racing with learning.

Two-phase system: map the track during practice, race with per-lap learning.
See docs/superpowers/specs/2026-03-19-px-race-design.md for full spec.
"""
from __future__ import annotations


def clamp(v: float, lo: float, hi: float) -> float:
    return max(lo, min(hi, v))


class PDController:
    """Proportional-Derivative controller with output clamping."""

    def __init__(
        self,
        kp: float,
        kd: float,
        output_min: float = -30.0,
        output_max: float = 30.0,
    ):
        self.kp = kp
        self.kd = kd
        self.output_min = output_min
        self.output_max = output_max
        self._prev_error: float | None = None

    def update(self, error: float, dt: float) -> float:
        p = self.kp * error
        if self._prev_error is not None and dt > 0:
            d = self.kd * (error - self._prev_error) / dt
        else:
            d = 0.0
        self._prev_error = error
        return clamp(p + d, self.output_min, self.output_max)

    def reset(self) -> None:
        self._prev_error = None
```

```python
def normalize_grayscale(
    readings: list[float],
    track_ref: list[float],
    barrier_ref: list[float],
) -> list[float]:
    """Normalize grayscale readings to 0.0 (track) – 1.0 (barrier)."""
    result = []
    for raw, t, b in zip(readings, track_ref, barrier_ref):
        span = b - t
        if span == 0:
            result.append(0.0)
        else:
            result.append(clamp((raw - t) / span, 0.0, 1.0))
    return result


def compute_edge_error(gs_norm: list[float]) -> float:
    """Compute edge error from normalized grayscale. Positive = drifting right.

    This is an error signal for the PD controller, not a steering angle.
    The PD controller converts positive error → negative steering (steer left).
    """
    return gs_norm[2] - gs_norm[0]
```

```python
class GateDetector:
    """Detect start/finish gate from grayscale deltas.

    Triggers on 2-of-3 sensors showing delta > threshold.
    Temporal confirmation: if only 1 triggers, waits up to confirm_frames
    for a 2nd. Debounce prevents double-counting.
    """

    def __init__(self, threshold: float, debounce_s: float = 3.0, confirm_frames: int = 3):
        self.threshold = threshold
        self.debounce_s = debounce_s
        self.confirm_frames = confirm_frames
        self._last_trigger_t: float = -999.0
        self._pending_count: int = 0  # sensors triggered in pending window
        self._pending_frames: int = 0  # frames since first pending trigger

    def update(self, prev_gs: list[float], gs: list[float], t: float) -> bool:
        """Check if gate was crossed. Returns True on detection."""
        # Debounce check
        if (t - self._last_trigger_t) < self.debounce_s:
            self._pending_count = 0
            self._pending_frames = 0
            return False

        # Count sensors with delta > threshold this frame
        triggered_this_frame = sum(
            1 for p, c in zip(prev_gs, gs) if abs(c - p) > self.threshold
        )

        if self._pending_count > 0:
            self._pending_frames += 1
            self._pending_count = max(self._pending_count, triggered_this_frame)

            if self._pending_count >= 2:
                self._last_trigger_t = t
                self._pending_count = 0
                self._pending_frames = 0
                return True

            if self._pending_frames > self.confirm_frames:
                # Confirmation window expired
                self._pending_count = 0
                self._pending_frames = 0

        if triggered_this_frame >= 2:
            self._last_trigger_t = t
            self._pending_count = 0
            self._pending_frames = 0
            return True
        elif triggered_this_frame == 1:
            self._pending_count = 1
            self._pending_frames = 0

        return False
```

```python
import json
from pathlib import Path


class TrackProfile:
    """Ordered list of track segments with persistence."""

    def __init__(self):
        self.segments: list[dict] = []
        self.map_speed: int = 20
        self.calibration_v: float = 0.0
        self.lap_duration_s: float = 0.0
        self.track_width_cm: float = 0.0
        self.lap_history: list[dict] = []

    def add_segment(
        self,
        seg_type: str,
        duration_s: float,
        width_left_cm: float,
        width_right_cm: float,
        sonar_center_cm: float,
        gs_signature: list[float],
    ) -> None:
        seg = {
            "id": len(self.segments),
            "type": seg_type,
            "duration_s": round(duration_s, 2),
            "width_left_cm": round(width_left_cm, 1),
            "width_right_cm": round(width_right_cm, 1),
            "sonar_center_cm": round(sonar_center_cm, 1),
            "race_speed": 28 if seg_type.startswith("turn") else 45,
            "steer_bias": 0,
            "entry_speed": 28 if seg_type.startswith("turn") else 45,
            "brake_before_s": 0.3 if seg_type.startswith("turn") else 0.0,
            "gs_signature": [round(v, 1) for v in gs_signature],
        }
        if seg_type.startswith("turn"):
            # Estimate steer bias from width imbalance
            diff = width_right_cm - width_left_cm
            seg["steer_bias"] = round(clamp(diff * 0.5, -30, 30), 1)
        self.segments.append(seg)

    def save(self, path: Path) -> None:
        data = {
            "mapped_at": "",
            "map_speed": self.map_speed,
            "calibration_v": self.calibration_v,
            "lap_duration_s": self.lap_duration_s,
            "track_width_cm": self.track_width_cm,
            "segments": self.segments,
            "lap_history": self.lap_history,
        }
        # Use path.write_text here; upgraded to atomic_write in Task 8
        # when pxh.state import is added for RaceController
        path.write_text(json.dumps(data, indent=2))

    @classmethod
    def load(cls, path: Path) -> "TrackProfile":
        data = json.loads(path.read_text())
        tp = cls()
        tp.map_speed = data.get("map_speed", 20)
        tp.calibration_v = data.get("calibration_v", 0.0)
        tp.lap_duration_s = data.get("lap_duration_s", 0.0)
        tp.track_width_cm = data.get("track_width_cm", 0.0)
        tp.segments = data.get("segments", [])
        tp.lap_history = data.get("lap_history", [])
        return tp


def classify_segment(
    left_cm: float,
    right_cm: float,
    center_cm: float,
    track_width_cm: float,
) -> str:
    """Classify a sensor reading as straight, turn_left, or turn_right."""
    half_width = track_width_cm / 2
    threshold = half_width * 0.20  # 20% of half-width
    imbalance = abs(left_cm - right_cm)
    center_close = center_cm < track_width_cm

    if imbalance > threshold and center_close:
        return "turn_left" if left_cm < right_cm else "turn_right"
    return "straight"
```

```python
import time


SERVO_SETTLE_S = 0.15  # 150ms — matches px-wander


def safe_ping(px, retries: int = 1) -> float | None:
    """Read sonar with I2C retry. Returns cm or None."""
    for attempt in range(1 + retries):
        try:
            return px.get_distance()
        except OSError:
            if attempt < retries:
                time.sleep(0.03)
    return None


def safe_grayscale(px, retries: int = 1) -> list[float] | None:
    """Read grayscale with I2C retry. Returns [left, center, right] or None."""
    for attempt in range(1 + retries):
        try:
            return px.get_grayscale_data()
        except OSError:
            if attempt < retries:
                time.sleep(0.01)
    return None

    Pan servo returns to 0 after scan.

        px.set_cam_pan_angle(angle)
        if settle_s > 0:
            time.sleep(settle_s)

    px.set_cam_pan_angle(0)
```

```python
MIN_RACE_SPEED = 5
MAX_SPEED_DELTA = 5


def apply_lap_learning(
    segment: dict,
    actual: dict,
    speed_ratio: float,
) -> dict:
    """Apply per-lap learning to a segment. Returns updated segment copy.

    speed_ratio = current_v / calibration_v (for battery compensation).
    actual = {"duration_s": float, "wall_clips": int, "obstacle": bool}
    """
    seg = dict(segment)  # shallow copy

    # Update duration (scaled by speed ratio for battery compensation)
    if speed_ratio > 0:
        seg["duration_s"] = round(actual["duration_s"] / speed_ratio, 2)

    if actual.get("obstacle"):
        # Obstacle — not the track's fault, no speed change
        return seg

    if actual["wall_clips"] > 0:
        # Wall clip — reduce speed, increase brake time
        seg["race_speed"] = max(MIN_RACE_SPEED, seg["race_speed"] - MAX_SPEED_DELTA)
        seg["entry_speed"] = min(seg["entry_speed"], seg["race_speed"])
        if seg.get("brake_before_s", 0) > 0:
            seg["brake_before_s"] = round(seg["brake_before_s"] + 0.1, 2)
    else:
        # Clean pass — bump speed (capped at +5 per lap per spec)
        seg["race_speed"] = seg["race_speed"] + 3
        seg["entry_speed"] = seg["race_speed"]

    # Note: max_speed cap applied by caller (RaceController) after this returns
    return seg
```

```python
def estop_threshold(speed: float) -> float:
    """Speed-dependent e-stop distance in cm."""
    return max(8.0, speed * 0.3)


def check_estop(sonar_cm: float | None, speed: float) -> bool:
    """Check if emergency stop should trigger."""
    if sonar_cm is None:
        return True  # sensor failure = stop
    return sonar_cm < estop_threshold(speed)


def check_edge_guard(
    gs_norm: list[float], threshold: float = 0.7
) -> tuple[bool, float]:
    """Check if any grayscale sensor is near barrier.

    Returns (triggered, steer_correction).
    steer_correction is negative = steer left, positive = steer right.
    """
    left, _center, right = gs_norm
    if left > threshold:
        return True, 15.0  # steer right, away from left barrier
    if right > threshold:
        return True, -15.0  # steer left, away from right barrier
    return False, 0.0


class StuckDetector:
    """Detect if the car is stuck (sonar unchanged for timeout_s)."""

    def __init__(self, timeout_s: float = 2.0, tolerance_cm: float = 3.0):
        self.timeout_s = timeout_s
        self.tolerance_cm = tolerance_cm
        self._last_change_t: float = 0.0
        self._last_cm: float | None = None

    def update(self, sonar_cm: float | None, t: float) -> None:
        if sonar_cm is None:
            return
        if self._last_cm is None or abs(sonar_cm - self._last_cm) > self.tolerance_cm:
            self._last_cm = sonar_cm
            self._last_change_t = t
        # If within tolerance, don't update _last_change_t → stuck timer ticks

    def is_stuck(self, t: float) -> bool:
        return (t - self._last_change_t) > self.timeout_s
```

```python
    def run_map(self, max_iterations: int = 0) -> None:
        """Run mapping lap. Builds self.profile from sensor data."""
        cal = self.calibration
        if not cal.get("track_ref") or not cal.get("barrier_ref"):
            raise RuntimeError("Calibration required before mapping")

        gate = GateDetector(
            threshold=cal.get("gate_threshold", 50),
            debounce_s=3.0,
            confirm_frames=3,
        )
        pd_sonar = PDController(kp=0.5, kd=0.2, output_min=-self.DIR_MAX, output_max=self.DIR_MAX)
        samples: list[dict] = []

        track_width = cal.get("track_width_cm", 88)
        prev_gs = safe_grayscale(self.px) or cal["track_ref"]

        # Warmup pings
        for _ in range(3):
            safe_ping(self.px)

        start_t = time.monotonic()
        iteration = 0

        while self._running:
            now = time.monotonic()
            elapsed = now - start_t

            gs = safe_grayscale(self.px) or prev_gs

            # Sonar sweep (3-point)
            left, right = quick3_scan(self.px, settle_s=0.0 if self.dry else SERVO_SETTLE_S)
            center = safe_ping(self.px)

            # Steer to stay centered
            if left is not None and right is not None:
                error = (right - left) if (right and left) else 0.0
                steer = pd_sonar.update(error, dt=0.75)
            else:
                steer = 0.0

            steer = clamp(steer, -self.DIR_MAX, self.DIR_MAX)

            if not self.dry:
                self.px.set_dir_servo_angle(steer)
                self.px.forward(self.MAP_SPEED)

            # Record sample
            samples.append({
                "t": round(elapsed, 3),
                "steer": round(steer, 1),
                "sonar_left": left,
                "sonar_center": center,
                "sonar_right": right,
                "gs": list(gs),
                "seg_type": classify_segment(
                    left or 50, right or 50, center or 100, track_width
                ),
            })

            # Gate detection
            if gate.update(prev_gs, gs, elapsed):
                break

            prev_gs = gs
            iteration += 1
            if max_iterations and iteration >= max_iterations:
                break

        if not self.dry:
            self.px.stop()

        # Compress samples into profile
        self.profile = self._compress_samples(samples, track_width)
        self.profile.map_speed = self.MAP_SPEED
        self.profile.track_width_cm = track_width
        self.profile.lap_duration_s = samples[-1]["t"] if samples else 0

        # Save
        self.profile.save(self.state_dir / "race_track.json")

    def _compress_samples(self, samples: list[dict], track_width: float) -> TrackProfile:
        """Compress raw sensor samples into segments."""
        profile = TrackProfile()
        if not samples:
            return profile

        current_type = samples[0]["seg_type"]
        seg_start = 0

        def _flush_segment(start: int, end: int, seg_type: str) -> None:
            seg_samples = samples[start:end]
            if not seg_samples:
                return
            duration = seg_samples[-1]["t"] - seg_samples[0]["t"]
            avg_left = _avg([s["sonar_left"] for s in seg_samples if s["sonar_left"]])
            avg_right = _avg([s["sonar_right"] for s in seg_samples if s["sonar_right"]])
            avg_center = _avg([s["sonar_center"] for s in seg_samples if s["sonar_center"]])
            avg_gs = [_avg([s["gs"][j] for s in seg_samples]) for j in range(3)]
            profile.add_segment(
                seg_type,
                duration_s=duration,
                width_left_cm=avg_left or 44,
                width_right_cm=avg_right or 44,
                sonar_center_cm=avg_center or 100,
                gs_signature=avg_gs,
            )

        for i, sample in enumerate(samples):
            if sample["seg_type"] != current_type:
                _flush_segment(seg_start, i, current_type)
                current_type = sample["seg_type"]
                seg_start = i

        # Flush final segment
        _flush_segment(seg_start, len(samples), current_type)

        return profile
```

```python
            dt = max(now - prev_loop_t, 0.001)  # measured, not hardcoded
            prev_loop_t = now

            segment = segments[current_seg_idx % len(segments)]

            # --- Sensors ---
            gs_raw = safe_grayscale(self.px)
            if gs_raw is None:
                i2c_failures += 1
                if i2c_failures >= 3:
                    if not self.dry:
                        self.px.stop()
                    break
                gs_raw = prev_gs
            else:
                i2c_failures = 0

            sonar_center = safe_ping(self.px)
            if sonar_center is None:
                i2c_failures += 1
                if i2c_failures >= 3:
                    if not self.dry:
                        self.px.stop()
                    break
            else:
                i2c_failures = 0

            # --- Grayscale edge steering ---
            gs_norm = normalize_grayscale(gs_raw, track_ref, barrier_ref)
            gs_steer_raw = compute_edge_error(gs_norm)
            gs_steer = pd_edge.update(gs_steer_raw, dt)

            # --- Edge guard (safety priority 2) ---
            edge_triggered, edge_correction = check_edge_guard(gs_norm)
            if edge_triggered:
                gs_steer = edge_correction
                speed = self.OBSTACLE_SPEED
                incidents += 1
                # Track wall clip for per-segment learning
                seg_key = current_seg_idx % len(segments)
                seg_actuals.setdefault(seg_key, {"duration_s": 0, "wall_clips": 0, "obstacle": False})
                seg_actuals[seg_key]["wall_clips"] += 1

            # --- Periodic Quick-3 ---
            if (now - last_quick3_t) > self.QUICK3_INTERVAL_S:
                left, right = quick3_scan(self.px, settle_s=0.0 if self.dry else SERVO_SETTLE_S)
                last_sonar_lr = (left, right)
                last_quick3_t = now

            # --- Sonar blend (age-weighted) ---
            sonar_age = now - last_quick3_t
            sonar_weight = max(0.0, 0.3 * (1.0 - sonar_age / 2.0))
            if last_sonar_lr[0] is not None and last_sonar_lr[1] is not None:
                sonar_error = (last_sonar_lr[1] - last_sonar_lr[0])
                sonar_steer = pd_sonar.update(sonar_error, dt)
            else:
                sonar_steer = 0.0
                sonar_weight = 0.0

            steer = (1.0 - sonar_weight) * gs_steer + sonar_weight * sonar_steer

            # --- Profile-based speed with transition blending ---
            if not edge_triggered:
                target_speed = min(segment.get("race_speed", 30), self.max_speed)
                target_steer_bias = segment.get("steer_bias", 0) if segment["type"].startswith("turn") else 0

                # Transition zone: blend from previous segment values
                if seg_elapsed < self.TRANSITION_ZONE_S and current_seg_idx > 0:
                    prev_seg = segments[(current_seg_idx - 1) % len(segments)]
                    blend = seg_elapsed / self.TRANSITION_ZONE_S  # 0→1 over transition
                    prev_speed = min(prev_seg.get("race_speed", 30), self.max_speed)
                    prev_bias = prev_seg.get("steer_bias", 0) if prev_seg["type"].startswith("turn") else 0
                    speed = prev_speed + (target_speed - prev_speed) * blend
                    target_steer_bias = prev_bias + (target_steer_bias - prev_bias) * blend
                else:
                    speed = target_speed

                # Brake before next segment
                next_seg_look = segments[(current_seg_idx + 1) % len(segments)]
                brake_before = next_seg_look.get("brake_before_s", 0)
                scaled_dur = segment["duration_s"] / max(speed / max(self.MAP_SPEED, 1), 0.5)
                if brake_before > 0 and (scaled_dur - seg_elapsed) < brake_before:
                    speed = min(speed, next_seg_look.get("entry_speed", 25))

                steer += target_steer_bias

            # --- E-stop (safety priority 1) ---
            if check_estop(sonar_center, speed):
                if not self.dry:
                    self.px.stop()
                    self.px.backward(20)
                    time.sleep(0.3)
                    self.px.stop()
                incidents += 1

            # --- Stuck detection (safety priority 5) ---
            stuck.update(sonar_center, now)
            if stuck.is_stuck(now):
                if not self.dry:
                    self.px.stop()
                break

            # --- Obstacle check (safety priority 3) ---
            if sonar_center is not None and not check_estop(sonar_center, speed):
                expected_center = segment.get("sonar_center_cm", 100)
                if sonar_center < expected_center * 0.5:
                    speed = self.OBSTACLE_SPEED

            # --- Apply steering and speed ---
            steer = clamp(steer, -self.DIR_MAX, self.DIR_MAX)
            if not self.dry:
                self.px.set_dir_servo_angle(steer)
                self.px.forward(speed)

            # --- Gate detection ---
            if gate.update(prev_gs, gs_raw, elapsed):
                lap_time = now - lap_start_t
                if lap_time < best_lap_s:
                    best_lap_s = lap_time

                # Per-lap learning: apply adjustments to profile
                battery_v = self._read_battery_voltage()
                speed_ratio = (battery_v / self.profile.calibration_v) if self.profile.calibration_v > 0 and battery_v > 0 else 1.0
                for seg_idx, seg in enumerate(segments):
                    seg_actual = seg_actuals.get(seg_idx, {"duration_s": seg["duration_s"], "wall_clips": 0, "obstacle": False})
                    segments[seg_idx] = apply_lap_learning(seg, seg_actual, speed_ratio)
                    segments[seg_idx]["race_speed"] = min(segments[seg_idx]["race_speed"], self.max_speed)
                seg_actuals.clear()

                # Log lap to race_log.jsonl
                self._append_race_log({
                    "lap": lap, "lap_time_s": round(lap_time, 2),
                    "best_lap_s": round(best_lap_s, 2) if best_lap_s < float("inf") else None,
                    "incidents": incidents, "battery_v": battery_v,
                })

                # Save updated profile
                self.profile.save(self.state_dir / "race_track.json")

                lap += 1
                lap_start_t = now
                gate_timeout_t = now
                current_seg_idx = 0
                seg_start_t = now
                pd_edge.reset()
                pd_sonar.reset()

                if max_laps and lap >= max_laps:
                    break

            # --- Timeout safety (layer 6) ---
            if (now - gate_timeout_t) > GATE_TIMEOUT_S:
                if not self.dry:
                    self.px.stop()
                break  # assume lost

            # --- Battery safety (layer 7) ---
            battery_v = self._read_battery_voltage()
            if battery_v and battery_v < 7.0:  # ~20% for 2S LiPo
                # Finish current lap then stop
                self.max_speed = min(self.max_speed, 25)

            # --- Segment advancement (sonar-primary, time-secondary) ---
            # Speed-ratio-scaled time estimate
            speed_ratio_now = speed / max(self.MAP_SPEED, 1)
            scaled_duration = segment["duration_s"] / max(speed_ratio_now, 0.5)

            # Sonar signature match: check if readings match next segment
            next_seg = segments[(current_seg_idx + 1) % len(segments)]
            sonar_matches_next = (
                sonar_center is not None
                and abs((sonar_center - next_seg.get("sonar_center_cm", 100))) < 20
            )

            # Orange corner landmark (if calibrated)
            orange_ref = cal.get("corner_ref")
            orange_detected = False
            if orange_ref:
                orange_dist = sum(abs(a - b) for a, b in zip(gs_raw, orange_ref))
                orange_detected = orange_dist < 100  # within calibrated range

            # Advance if: sonar matches next segment, OR time exceeded, OR orange landmark
            if sonar_matches_next and seg_elapsed > scaled_duration * 0.5:
                current_seg_idx += 1
                seg_start_t = now
            elif seg_elapsed > scaled_duration:
                current_seg_idx += 1
                seg_start_t = now
            elif orange_detected and next_seg["type"].startswith("turn"):
                current_seg_idx += 1
                seg_start_t = now

            # --- Lost recovery ---
            if seg_elapsed > scaled_duration * 3:
                # Way past expected segment — fall back to reactive wall-following
                speed = self.OBSTACLE_SPEED
                # Rely purely on grayscale edge avoidance until re-sync

            # --- Live telemetry ---
            if (now - last_telemetry_t) > self.TELEMETRY_INTERVAL_S:
                self._write_live_telemetry({
                    "ts": time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime()),
                    "mode": "race",
                    "lap": lap,
                    "segment": current_seg_idx % len(segments),
                    "speed": speed,
                    "steer": round(steer, 1),
                    "sonar_center_cm": sonar_center,
                    "gs": list(gs_raw),
                    "incidents": incidents,
                    "battery_v": self._read_battery_voltage(),
                    "lap_time_s": round(now - lap_start_t, 1),
                    "best_lap_s": round(best_lap_s, 1) if best_lap_s < float("inf") else None,
                })
                last_telemetry_t = now

            prev_gs = gs_raw
            iteration += 1
            if max_iterations and iteration >= max_iterations:
                break

        if not self.dry:
            self.px.stop()
        self._set_exploring(False)
```

```python
    _cached_battery_v: float = 0.0
    _battery_read_t: float = 0.0
    _BATTERY_CACHE_S: float = 15.0  # re-read every 15s, not every loop iteration

    def _read_battery_voltage(self) -> float:
        """Read current battery voltage from state/battery.json (cached)."""
        now = time.monotonic()
        if (now - self._battery_read_t) < self._BATTERY_CACHE_S:
            return self._cached_battery_v
        try:
            path = self.state_dir / "battery.json"
            if path.exists():
                data = json.loads(path.read_text())
                self._cached_battery_v = float(data.get("voltage", 0))
        except (json.JSONDecodeError, OSError):
            pass
        self._battery_read_t = now
        return self._cached_battery_v

    def _set_exploring(self, active: bool) -> None:
        """Set exploring.json to prevent/allow px-alive restart."""
        path = self.state_dir / "exploring.json"
        data = {"active": active, "pid": os.getpid()} if active else {"active": False}
        try:
            atomic_write(path, json.dumps(data))
        except OSError:
            pass

    def _append_race_log(self, entry: dict) -> None:
        """Append a lap entry to race_log.jsonl."""
        path = self.state_dir / "race_log.jsonl"
        entry["ts"] = time.strftime("%Y-%m-%dT%H:%M:%SZ", time.gmtime())
        try:
            with open(path, "a") as f:
                f.write(json.dumps(entry) + "\n")
        except OSError:
            pass
```

```python
import argparse


def main() -> int:
    parser = argparse.ArgumentParser(description="PiCar-X autonomous track racer")
    group = parser.add_mutually_exclusive_group()
    group.add_argument("--calibrate", action="store_true", help="On-site sensor calibration")
    group.add_argument("--map", action="store_true", help="Practice lap — map the track")
    group.add_argument("--race", action="store_true", help="Timed race using stored profile")
    group.add_argument("--status", action="store_true", help="Print current profile summary")
    parser.add_argument("--laps", type=int, default=0, help="Number of laps (0=unlimited)")
    parser.add_argument("--max-speed", type=int, default=50, help="Max speed PWM (cap 60)")
    parser.add_argument("--dry-run", action="store_true", help="No motor output")

    args = parser.parse_args()
    dry = args.dry_run or os.environ.get("PX_DRY", "0") != "0"

    state_dir = Path(os.environ.get("PX_STATE_DIR", Path(os.environ.get("PROJECT_ROOT", ".")) / "state"))

    if args.status:
        prof_path = state_dir / "race_track.json"
        if not prof_path.exists():
            print("No profile found. Run --map first.")
            return 0
        profile = TrackProfile.load(prof_path)
        print(f"Track profile: {len(profile.segments)} segments, "
              f"lap duration: {profile.lap_duration_s:.1f}s, "
              f"width: {profile.track_width_cm:.0f}cm")
        for seg in profile.segments:
            print(f"  [{seg['id']}] {seg['type']:12s} {seg['duration_s']:.1f}s "
                  f"speed={seg['race_speed']} steer={seg.get('steer_bias', 0)}")
        return 0

    # Need Picarx for everything else
    if dry:
        from unittest.mock import MagicMock
        px = MagicMock()
        px.get_grayscale_data.return_value = [400, 410, 405]
        px.get_distance.return_value = 90.0
    else:
        from picarx import Picarx
        px = Picarx()

    try:
        rc = RaceController(px=px, state_dir=state_dir, dry=dry, max_speed=args.max_speed)

        if args.calibrate:
            print("=== Calibration ===")
            print("Place SPARK on the dark gray track surface, then press Enter.")
            input()
            rc._calibrate_surface("track", px)
            print(f"Track ref: {rc.calibration['track_ref']}")

            print("Place SPARK sensors over the light gray barrier, then press Enter.")
            input()
            rc._calibrate_surface("barrier", px)
            print(f"Barrier ref: {rc.calibration['barrier_ref']}")

            # Track width from sonar
            print("Point SPARK straight down a track section, then press Enter.")
            input()
            left, right = quick3_scan(px)
            if left and right:
                rc.calibration["track_width_cm"] = round(left + right, 1)
                print(f"Track width: {rc.calibration['track_width_cm']}cm")

            rc.calibration["gate_threshold"] = 50  # default, tune on-site
            rc.save_calibration()
            print(f"Calibration saved to {state_dir / 'race_calibration.json'}")

        elif args.map:
            print("=== Map Mode ===")
            rc.run_map()
            if rc.profile:
                print(f"Mapped {len(rc.profile.segments)} segments, "
                      f"lap: {rc.profile.lap_duration_s:.1f}s")

        elif args.race:
            print("=== Race Mode ===")
            rc.run_race(max_laps=args.laps)
            print("Race complete.")

        else:
            parser.print_help()

    finally:
        if not dry and hasattr(px, "stop"):
            px.stop()

    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```
