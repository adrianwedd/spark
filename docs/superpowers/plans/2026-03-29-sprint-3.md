```bash
#!/usr/bin/env bash
# bin/px-race-check — Pre-race hardware validation
set -euo pipefail
SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
source "$SCRIPT_DIR/px-env"

DRY=0
for arg in "$@"; do
    case "$arg" in
        --dry-run) DRY=1 ;;
    esac
done

exec /usr/bin/python3 - "$DRY" <<'PYEOF'
import sys, json, os, pathlib

dry = sys.argv[1] == "1"
state_dir = pathlib.Path(os.environ.get("PX_STATE_DIR", "state"))

checks = {}

# 1. Calibration file exists and has surface refs
cal_file = state_dir / "race_calibration.json"
if cal_file.exists():
    try:
        cal = json.loads(cal_file.read_text())
        if cal.get("track_ref") and cal.get("barrier_ref"):
            checks["calibration"] = "ok"
        else:
            checks["calibration"] = "incomplete"
    except Exception:
        checks["calibration"] = "corrupt"
else:
    checks["calibration"] = "missing"

# 2. Track profile exists
track_file = state_dir / "race_track.json"
if track_file.exists():
    try:
        profile = json.loads(track_file.read_text())
        segs = profile.get("segments", [])
        checks["profile"] = f"ok ({len(segs)} segments)" if segs else "empty"
    except Exception:
        checks["profile"] = "corrupt"
else:
    checks["profile"] = "missing"

# 3. Battery check
batt_file = state_dir / "battery.json"
if batt_file.exists():
    try:
        batt = json.loads(batt_file.read_text())
        pct = batt.get("battery_pct", 0)
        checks["battery"] = f"ok ({pct}%)" if pct > 30 else f"low ({pct}%)"
    except Exception:
        checks["battery"] = "unknown"
else:
    checks["battery"] = "unknown"

# 4. I2C bus (skip in dry mode)
if dry:
    checks["i2c"] = "skipped (dry)"
else:
    try:
        import smbus2
        bus = smbus2.SMBus(1)
        bus.read_byte(0x14)  # PCA9685 default address
        bus.close()
        checks["i2c"] = "ok"
    except Exception as e:
        checks["i2c"] = f"error: {e}"

# 5. px-alive status (should be running — we'll yield it before race)
if dry:
    checks["px_alive"] = "skipped (dry)"
else:
    import subprocess
    r = subprocess.run(["systemctl", "is-active", "px-alive"], capture_output=True, text=True)
    checks["px_alive"] = r.stdout.strip()

ready = (
    checks.get("calibration", "").startswith("ok")
    and checks.get("battery", "").startswith(("ok", "unknown"))
    and not checks.get("i2c", "").startswith("error")
)

result = {"checks": checks, "ready": ready}
print(json.dumps(result))
PYEOF
```

## P1 — Mood Diversity Fix

The problem: 94% of thoughts land in 3 moods (contemplative/content/peaceful). Root causes:

1. **Default fallback is "content"** — invalid/missing mood always → content (mind.py:2434)
2. **Mood momentum ALPHA=0.3** — 70% inertia locks in patterns
3. **No mood-to-angle mapping** — LLM infers mood from angle alone; "contemplative" is the safe token
4. **Prompt lists moods without weighting** — Haiku gravitates to safe options

import subprocess

def _run_story(action, extra_env=None, **params):
    """Helper to run tool-story with given action and params."""
    env = os.environ.copy()
    env["PX_DRY"] = "1"
    env["PX_BYPASS_SUDO"] = "1"
    if extra_env:
        env.update(extra_env)
    env["TOOL_ACTION"] = action
    for k, v in params.items():
        env[f"TOOL_{k.upper()}"] = str(v)
    r = subprocess.run(
        ["bash", "bin/tool-story"],
        capture_output=True, text=True, env=env,
        cwd=os.environ.get("PROJECT_ROOT", "."),
    )
    return r, json.loads(r.stdout.strip().split("\n")[-1])

@app.get("/api/v1/public/budget")
async def public_budget():
    """Claude session budget status — unauthenticated."""
    try:
        from pxh.claude_session import DAILY_CAP, _load_session_log, _today_entries
        log_entries = _load_session_log()
        today = _today_entries(log_entries)
        sessions = [
            {
                "ts": e.get("ts"),
                "type": e.get("type"),
                "model": e.get("model"),
                "duration_s": e.get("duration_s"),
                "outcome": e.get("outcome"),
            }
            for e in today
        ]
        return {
            "daily_cap": DAILY_CAP,
            "used_today": len(today),
            "remaining": max(0, DAILY_CAP - len(today)),
            "sessions": sessions,
        }
    except Exception:
        return {"daily_cap": 8, "used_today": 0, "remaining": 8, "sessions": []}

function updateBudget(data) {
    const el = document.getElementById('budget-display');
    if (!el || !data) return;
    const used = data.used_today || 0;
    const cap = data.daily_cap || 8;
    const remaining = data.remaining || 0;

    // Color: green if >50% remaining, yellow if >25%, red if <=25%
    let color = 'var(--mood-content)';
    if (remaining <= 2) color = 'var(--mood-anxious)';
    else if (remaining <= 4) color = 'var(--mood-alert)';

    // Clear and rebuild safely
    el.textContent = '';
    const span = document.createElement('span');
    span.style.color = color;
    span.textContent = used + '/' + cap;
    el.appendChild(span);

    if (data.sessions && data.sessions.length > 0) {
        const last = data.sessions[data.sessions.length - 1];
        const small = document.createElement('small');
        small.textContent = ' (last: ' + (last.type || '?') + ')';
        el.appendChild(small);
    }
}
